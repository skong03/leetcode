public class Solution {
    public boolean isNumber(String s) {
        // Start typing your Java solution below
        // DO NOT write main() function
        char[] mychar= s.toCharArray();
        boolean dot=true;
        boolean shuzi=false;
        boolean pos=true;
        boolean neg=true;
        boolean space=true;
        boolean flag=true;
        boolean space1=true;
        
        if(s.length()==0)
        return false;
        
        for(int i =0; i< mychar.length; i++)
        {
            
            if(mychar[i]=='+'&& pos==true)
            {
                space1=false;
                pos=false;
                continue;
            }
            
            if(mychar[i]=='-'&& neg==true)
            {
                space1=false;
                neg=false;
                continue;
            }
            
            if(mychar[i]==' ' && !shuzi &&space1)
            {
                continue;
                //space =true;
            }
            
            if(mychar[i]==' '&& shuzi && space1)
            {
                flag=false;
                continue;
            }
            
            if(mychar[i]=='.'&&dot)//当出现数字以后才能出现。
            {
                dot=false;
                continue;
            }
            
            if(mychar[i]>='0'&& mychar[i]<='9' &&flag)
            {
                shuzi=true;
                //space=false;
                continue;
            }
            
            if(mychar[i]=='e'&& i>=1&& mychar[i-1]!='.' && i+1< mychar.length && mychar[i+1]!='.')
            {
                continue;
            }
            
            return false;
        }
        
        return (true&&shuzi);
        
        
    }
}
